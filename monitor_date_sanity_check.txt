# ── Patch for openclaw-monitor.php ──────────────────────────────────────────
#
# PROBLEM: tonight's Larimer Lounge incident wasn't just a scraper bug --
# Monitor's Fire results page showed "104 events parsed, 104/104 have an
# image" looking completely healthy, with nothing visually distinguishing
# "104 legitimately different events" from "104 events that are all
# secretly dated today due to a bug." The Post button looked exactly as
# safe to click either way.
#
# THE FIX: after parsing the fired batch, check what fraction of events
# share the exact same date. If it's suspiciously high, show a loud
# warning ABOVE the Post button instead of the normal green summary --
# doesn't block posting (some sources legitimately have many same-day
# showtimes), but makes "something might be wrong" impossible to miss.
#
# ── In run_scrape() (or wherever the Fire results page is rendered),
#    right after $events is decoded from the JSON payload and $count is
#    computed, add: ───────────────────────────────────────────────────────

        // Detect suspiciously uniform dates -- a strong signal of a
        // date-parsing bug (e.g. silently defaulting to "today") rather
        // than a real source where many events happen to share a date.
        $date_counts = [];
        foreach ( $events as $e ) {
            $d = substr( $e['start_date'] ?? '', 0, 10 ); // YYYY-MM-DD only
            if ( $d === '' ) continue;
            $date_counts[ $d ] = ( $date_counts[ $d ] ?? 0 ) + 1;
        }
        $max_same_date = $date_counts ? max( $date_counts ) : 0;
        $today         = date( 'Y-m-d' );
        $same_date_pct = $count > 0 ? ( $max_same_date / $count ) : 0;

        // Threshold: more than 80% of events sharing one date, AND that
        // date is today, is the specific pattern this bug produces.
        // (Deliberately checks "is today" too -- a source with 80+
        // events all on, say, a single festival day next month is a
        // real and legitimate pattern; all-dated-TODAY across a random
        // Fire click is not.)
        $looks_like_date_bug = ( $count >= 10 && $same_date_pct >= 0.80
                                  && isset( $date_counts[ $today ] )
                                  && $date_counts[ $today ] === $max_same_date );

# ── Then in the HTML, replace the existing summary block: ──────────────────
#
#   <div class="summary">
#       <div><b><?= $count ?></b> events parsed</div>
#       ...
#   </div>
#
# with:

        <?php if ( $looks_like_date_bug ): ?>
        <div class="summary" style="background:#fdecea; border:2px solid #c0392b;">
            <div style="color:#c0392b; font-weight:700; font-size:16px; margin-bottom:8px;">
                ⚠ <?= $date_counts[ $today ] ?> / <?= $count ?> events are dated TODAY (<?= esc_html( $today ) ?>)
            </div>
            <div style="color:#c0392b;">
                This usually means the scraper couldn't find a real date on the page
                and is defaulting to "today" instead. Check a few of these events
                against the actual venue website before posting -- if their real
                dates are spread across different days, this batch is NOT safe to post yet.
            </div>
        </div>
        <?php endif; ?>

        <div class="summary">
            <div><b><?= $count ?></b> events parsed</div>
            <div><?= $with_image === $count ? '<span class="ok">✓</span>' : '<span class="warn">⚠</span>' ?> <b><?= $with_image ?> / <?= $count ?></b> have an image</div>
            <div><b><?= $with_cost ?> / <?= $count ?></b> have a price listed (blank is normal for free events)</div>
        </div>

# Note: this check is intentionally conservative (>=10 events, >=80%
# same date, AND that date is specifically today) to avoid false-alarming
# on small batches or legitimate single-day-festival sources. It also
# doesn't block the Post button -- it's a visibility fix, not a hard
# stop, since you're the one who knows whether a given source legitimately
# has this pattern.
